/* ---------- UI wiring: populate grid from PROJECTS ---------- */
const grid = document.getElementById('grid');
function populateGrid(){
  grid.innerHTML = '';
  PROJECTS.forEach((p, i) => {
    const card = document.createElement('article');
    card.className = 'card reveal';
    card.dataset.category = p.category;
    card.dataset.index = i;
    card.tabIndex = 0;
    const img = document.createElement('img');
    img.className = 'thumb';
    img.src = p.thumb;
    img.alt = p.title;
    const meta = document.createElement('div');
    meta.className = 'meta';
    meta.innerHTML = `<h3>${p.title}</h3><p class="muted">${p.role} · ${p.tech.split(',')[0]}</p>`;
    card.appendChild(img); card.appendChild(meta);
    grid.appendChild(card);
  });
}
populateGrid();

/* ---------- Filter behavior ---------- */
const filters = document.querySelectorAll('.filter');
filters.forEach(f => f.addEventListener('click', () => {
  filters.forEach(x=>x.classList.remove('active'));
  f.classList.add('active');
  const filter = f.dataset.filter;
  document.querySelectorAll('.card').forEach(card=>{
    if(filter === 'all' || card.dataset.category === filter) card.style.display = '';
    else card.style.display = 'none';
  });
}));

/* ---------- Reveal on scroll ---------- */
function revealOnScroll(){
  document.querySelectorAll('.reveal').forEach(el=>{
    const top = el.getBoundingClientRect().top;
    if(top < window.innerHeight - 80) el.classList.add('show');
  });
}
window.addEventListener('scroll', revealOnScroll);

/* ---------- Modal (lightbox) ---------- */
const modal = document.getElementById('modal');
const modalMedia = document.getElementById('modalMedia');
const modalTitle = document.getElementById('modalTitle');
const modalMeta = document.getElementById('modalMeta');
const modalDesc = document.getElementById('modalDesc');
const modalTech = document.getElementById('modalTech');
const downloadMedia = document.getElementById('downloadMedia');
let currentIndex = 0;

function openModal(index){
  const p = PROJECTS[index];
  currentIndex = index;
  // media area
  modalMedia.innerHTML = '';
  if(p.type === 'video'){
    const v = document.createElement('video');
    v.src = p.src; v.controls = true; v.autoplay = true; v.style.maxWidth='100%';
    v.setAttribute('playsinline','');
    modalMedia.appendChild(v);
    downloadMedia.href = p.src; downloadMedia.download = '';
  } else {
    const img = document.createElement('img');
    img.src = p.src; img.alt = p.title; img.style.maxWidth='100%';
    modalMedia.appendChild(img);
    downloadMedia.href = p.src; downloadMedia.download = '';
  }
  modalTitle.textContent = p.title;
  modalMeta.textContent = `${p.role} · ${p.category.toUpperCase()}`;
  modalDesc.textContent = p.desc;
  modalTech.textContent = p.tech;
  modal.classList.add('show');
  modal.setAttribute('aria-hidden','false');
}
function closeModal(){ 
  // pause any video
  const v = modalMedia.querySelector('video'); if(v){ try{ v.pause(); v.currentTime = 0 }catch(e){} }
  modal.classList.remove('show'); modal.setAttribute('aria-hidden','true');
}

document.getElementById('modalClose').addEventListener('click', closeModal);
modal.addEventListener('click', (e)=>{ if(e.target === modal) closeModal(); });
window.addEventListener('keydown', (e)=>{ if(e.key === 'Escape') closeModal(); });

function handleCvFile(e){
  const f = e.target.files && e.target.files[0];
  if(!f) return;
  if(f.type !== 'application/pdf') return alert('Please upload a PDF file.');
  const url = URL.createObjectURL(f);
  cvPreview.innerHTML = '';
  const iframe = document.createElement('iframe');
  iframe.src = url; iframe.title = 'CV preview';
  cvPreview.appendChild(iframe);
  downloadCvBtn.style.display = 'inline-block';
  downloadCvBtn.href = url; downloadCvBtn.download = f.name || 'CV.pdf';
  downloadCvBtn.textContent = 'Download Uploaded CV';
  downloadCvHost.style.pointerEvents = 'auto'; downloadCvHost.style.opacity = 1; downloadCvHost.href = url;
  downloadCvHost.textContent = 'Download CV (hosted)';
}
